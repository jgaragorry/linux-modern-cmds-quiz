Cheat-sheet rápido

rg (ripgrep)

rg -n -P --hidden --no-ignore -g '!.git/' PATTERN

fd

fd -H -I -t f REGEX --strip-cwd-prefix

bat

bat --line-range A:B --style=numbers,changes --paging=never file

eza

eza -lah --git --group-directories-first -s modified -r

duf / dust

duf        # reemplazo de df
dust -d 2  # reemplazo de du

zoxide + fzf

j(){ cd "$(zoxide query -l | fzf)" || return; }

hyperfine

hyperfine -w 5 -r 10 'cmd1' 'cmd2' --export-json out.json

git-delta

git config --global core.pager delta
